灿吧销量统计

by 四层巧克力 | 预计统计每日18:00(GMT+8)时的数据

数据来源于灿吧置顶链接，如有遗漏私信我改

仅统计公开销量数据的链接，部分微店链接不显示销量所以没有计入统计

library("tidyverse")
library('extrafont')
library('grid')
everydaydata <- read.csv("~/Desktop/拉表数据/6.22/everydaydata.csv")
dataset1 = read.csv('everydaydata.csv')
totalset = dataset1[,c('date', 'total')]

总销量涨幅

ggplot(totalset, aes(x = date, y = total, group = 1 )) + geom_bar(stat = "identity", fill = "#fad65c") + geom_point(color = "#FAB612") + geom_line(color = "#FAB612", linewidth = 1) + geom_text(aes(label = total), vjust = -0.5, col = "#7D4402") + labs(title = "灿吧中输涨幅", x = '日期', y = "总销量") + theme(text = element_text(family = "STHeiti"))

销量分布

colors = c("#FFDE5A", "#FFC944", "#FFA646", "#FF8F30")
data1 = data.frame(category = c("pb", "vm", "poster", "qr"), value = c(5060, 2685, 5877, 4601))
ggplot(data1, aes(x = "", y = value, fill = category)) + geom_bar(stat = "identity", width = 1) + coord_polar(theta = "y") + labs(x = '', y = '', title = "6.22")+ geom_text(aes(label = value),position = position_stack(vjust = 0.5), col = "#7D4402") + theme_minimal() + theme(text = element_text(family = "STHeiti")) + scale_fill_manual(values = colors)
